    <script>
        const dropZone = document.getElementById('drop-zone');
        const terminal = document.getElementById('terminal');
        const preview = document.getElementById('preview-section');
        const hero = document.getElementById('hero-text');

        document.getElementById('file-input').addEventListener('change', function(e) {
            if (e.target.files.length > 0) startSequence(e.target.files[0].name);
        });

        async function startSequence(name) {
            dropZone.style.display = 'none';
            hero.style.display = 'none';
            terminal.style.display = 'block';
            
            const lines = [
                `> Ingesting ${name}...`,
                `> Generating Schema Hash...`,
                `> Constructing Target: Shopify_v2`,
                `> Executing Python Transformation...`,
                `> Success: 15,000 rows formatted.`
            ];

            for (let line of lines) {
                const div = document.createElement('div');
                div.textContent = line;
                document.getElementById('terminal-output').appendChild(div);
                await new Promise(r => setTimeout(r, 800));
            }

            await new Promise(r => setTimeout(r, 1000));
            terminal.style.display = 'none';
            preview.style.display = 'flex';
        }
    </script>
